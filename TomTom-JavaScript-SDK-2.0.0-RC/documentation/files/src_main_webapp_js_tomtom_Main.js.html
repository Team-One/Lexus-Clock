/**

define(["./lib/json2"], function() {

	// define the main TomTom namespace
	if (!window.tomtom) {
		window.tomtom = {};
	}
	
	window.tomtom.dom = {};
	window.tomtom.layers = {};
	window.tomtom.services = {};
	window.tomtom.controls = {};

	// IE doesn't have trim method for strings...
	if (typeof String.prototype.trim !== 'function') {
		String.prototype.trim = function() {
			return this.replace(/^\s+|\s+$/g, '');
		};
	}
	
	// again, IE, dumb
	if (!Array.prototype.indexOf) {
	    Array.prototype.indexOf = function (searchElement /*, fromIndex */ ) {
	        "use strict";
	        if (this == null) {
	            throw new TypeError();
	        }
	        var t = Object(this);
	        var len = t.length >>> 0;
	        if (len === 0) {
	            return -1;
	        }
	        var n = 0;
	        if (arguments.length > 0) {
	            n = Number(arguments[1]);
	            if (n != n) { // shortcut for verifying if it's NaN
	                n = 0;
	            } else if (n != 0 && n != Infinity && n != -Infinity) {
	                n = (n > 0 || -1) * Math.floor(Math.abs(n));
	            }
	        }
	        if (n >= len) {
	            return -1;
	        }
	        var k = n >= 0 ? n : Math.max(len - Math.abs(n), 0);
	        for (; k < len; k++) {
	            if (k in t && t[k] === searchElement) {
	                return k;
	            }
	        }
	        return -1;
	    };
	}
	
	if ( !Array.prototype.forEach ) {
	  Array.prototype.forEach = function(fn, scope) {
	    for(var i = 0, len = this.length; i < len; ++i) {
	      fn.call(scope, this[i], i, this);
	    }
	  };
	}
	
	if (!Array.prototype.some)
	{
	  Array.prototype.some = function(fun /*, thisp */)
	  {
	    "use strict";
	 
	    if (this == null)
	      throw new TypeError();
	 
	    var t = Object(this);
	    var len = t.length >>> 0;
	    if (typeof fun != "function")
	      throw new TypeError();
	 
	    var thisp = arguments[1];
	    for (var i = 0; i < len; i++)
	    {
	      if (i in t && fun.call(thisp, t[i], i, t))
	        return true;
	    }
	 
	    return false;
	  };
	}
	
	var toString = Object.prototype.toString,
    strArray = Array.toString();

	Array.isArray = Array.isArray || function (obj) {
	   return typeof obj == "object" && (toString.call(obj) == "[object Array]" || ("constructor" in obj && String(obj.constructor) == strArray));
	};
	
	/**
	 * Sets the image path that will be used throughout the SDK.
	 * @method tomtom.setImagePath
	 * @static
	 * @param path - The path to the directory where images are stored.
	 */
	window.tomtom.setImagePath = function(path) {
		
		// add a trailing slash if the path doesn't end in one
		if (path.lastIndexOf("/") != path.length - 1)
			path += "/";
		
		tomtom.baseImagePath = path;
		
	};
	
	// set the default image location
	tomtom.setImagePath("./images/");
	
	/**
	 * The base path for all API calls.
	 */
	window.tomtom.API_BASE_URL = "https://api.tomtom.com/";
	
	/**
	 * Contains build information for the TomTom JavaScript SDK.
	 * @property {Object} tomtom.releaseInfo
	 * @static
	 * @property {String} tomtom.releaseInfo.version The version information for the SDK.
	 * @property {String} tomtom.releaseInfo.build The build number for the SDK.

	window.tomtom.releaseInfo = {
		version: "@SDK_VERSION@",

	};
	
	/**
	 * Determines whether or not debug output should be displayed in the console.

	 * @static
	 */
	window.tomtom.isDebugMode = true;

	 */
