define([
        "../Main",
        "../Utils"
   ], 
    
    function(main, utils) {
	
		tomtom.layers.MapLayer = L.TileLayer.extend({
		
			/**
			 * The TomTom Basic map layer.  This layer displays the TomTom Map using the Map Toolkit's Raster Tile API.
			 * @constructor
			 * @class MapLayer
			 * @namespace tomtom.layers
			 * @constructor
			 * @extends L.TileLayer
			 * @param {Object} [options] Configuration options for the layer.
			 * @param {String} [options.apiKey] The Map Toolkit API that should be used with the Raster Tile API.  This will override the tomtom.apiKey property.
			 * @param {Number} [options.minZoom=0] Minimum zoom number.
			 * @param {Number} [options.maxZoom=18] Maximum zoom number.
			 * @param {Number} [options.tileSize=256] Tile size (width and height in pixels, assuming tiles are square).
			 * @param {String} [options.subdomains='abc'] Subdomains of the tile service. Can be passed in the form of one string (where each letter is a subdomain name) or an array of strings.
			 * @param {String} [options.errorTileUrl=''] URL to the tile image to show in place of the tile that failed to load.
			 * @param {String} [options.attribution=''] e.g. "© CloudMade" — the string used by the attribution control, describes the layer data.
			 * @param {Boolean} [options.tms=false] If true, inverses Y axis numbering for tiles (turn this on for TMS services).
			 * @param {Boolean} [options.continuousWorld=false] If set to true, the tile coordinates won't be wrapped by world width (-180 to 180 longitude) or clamped to lie within world height (-90 to 90). Use this if you use Leaflet for maps that don't reflect the real world (e.g. game, indoor or photo maps).
			 * @param {Boolean} [options.noWrap=false] If set to true, the tiles just won't load outside the world width (-180 to 180 longitude) instead of repeating.
			 * @param {Number} [options.zoomOffset=0] The zoom number used in tile URLs will be offset with this value.
			 * @param {Boolean} [options.zoomReverse=false] If set to true, the zoom number used in tile URLs will be reversed (maxZoom - zoom instead of zoom)
			 * @param {Number} [options.opacity=1.0] The opacity of the tile layer.
			 * @param {Number} [options.zIndex=null] The explicit zIndex of the tile layer. Not set by default.
			 * @param {Boolean} [options.unloadInvisibleTiles=depends] If true, all the tiles that are not visible after panning are removed (for better performance). true by default on mobile WebKit, otherwise false.
			 * @param {Boolean} [options.updateWhenIdle=depends] If false, new tiles are loaded during panning, otherwise only after it (for better performance). true by default on mobile WebKit, otherwise false.
			 * @param {Boolean} [options.detectRetina=false] If true and user is on a retina display, it will request four tiles of half the specified size and a bigger zoom level in place of one to utilize the high resolution.
			 * @param {Boolean} [options.reuseTiles=false] If true, all the tiles that are not visible after panning are placed in a reuse queue from which they will be fetched when new tiles become visible (as opposed to dynamically creating new ones). This will in theory keep memory usage low and eliminate the need for reserving new memory whenever a new tile is needed.
			 */
			initialize: function(options) {
				
				if (typeof(options) == "undefined")
					options = {};
				
				if (!options.apiKey)
					options.apiKey = tomtom.apiKey;
				
				// check to see if API proxying is enabled
				var url = utils.getServiceUrl("lbs/map/3/basic/1/{z}/{x}/{y}.png");
				
				// replace the host with a random host prefix
				if (typeof(tomtom.enableHostCycling) == "undefined" || tomtom.enableHostCycling) {
					url = url.replace(/https?:\/\//g, "http://{s}.");
				}
				
				// add the key if the proxy is disabled
				if (!tomtom.proxyUrl)
					url += "?key={apiKey}";
				
				L.TileLayer.prototype.initialize.apply(this, [url, options]);
				
			}
		    
		});
		
		return tomtom.layers.MapLayer;
		
	}
);
