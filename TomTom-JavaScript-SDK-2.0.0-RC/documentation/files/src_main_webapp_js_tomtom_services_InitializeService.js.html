define([
        "./BaseService",
        "../Utils",
        "../AjaxUtil"
    ], 
    
    function(base, utils, ajax) {
	
		tomtom.services.InitializeService = base.extend({
			
			/**
			 * A service which gets copyright information about the map toolkit..
			 * @constructor
			 * @class InitializeService
			 * @namespace tomtom.services
			 * @param {String} apiKey The API key used to access the Map Toolkit API.
			 */
			initialize: function(apiKey) {
				base.prototype.initialize.apply(this, arguments);
			},
			
			/**
			 * Gets copyright information for the Map Toolkit API.
			 * @method getCopyrightInfo
			 * @param {Object} [options] The options to pass to the initialize call.
			 * @param {String} [options.projection=EPSG900913] The projection to use for the points returned by the initialize call. Supported projections are EPSG4326 and EPSG900913.
			 * @param {Function} callback The method to call when the initialize call completes.
			 */
			getCopyrightInfo: function(options, callback) {
				
				// make the options parameter optional
				if (typeof(options) == "function") {
					callback = options;
					options = {};
				}
				
				// if an API key is specified, include it
				// an API key can be ommitted if an API proxy is being used
				if (this.apiKey)
					options.key = this.apiKey;
				
				// check to see if an API proxy is being used
				// if so, use a regular XHR, JSONP will not work with a proxy
				if (tomtom.proxyUrl) {
					ajax.getJSON({
						url: utils.getServiceUrl(INITIALIZE_URL, options),
						callback: callback
					});
				} else {
					ajax.getJSONP({
						url: utils.getServiceUrl(INITIALIZE_URL + "p", options),
						callback: callback,
						callbackParam: "jsonp"
					});
				}
				
			}
			
		});
		
		var INITIALIZE_URL = "lbs/services/initialize/3/json";
		
		return tomtom.services.InitializeService;
	}
);
