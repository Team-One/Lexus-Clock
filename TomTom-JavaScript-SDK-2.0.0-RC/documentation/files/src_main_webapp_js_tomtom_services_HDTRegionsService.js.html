File: src/main/webapp/js/tomtom/services/HDTRegionsService.js

define([
        "./BaseService",
        "../AjaxUtil",
        "../Utils"
    ], 
    
    function(base, ajax, utils) {
	
		tomtom.services.HDTRegionsService = base.extend({
			
			/**
			 * A service which gets information about overall traffic conditions in the world.
			 * @constructor
			 * @class HDTRegionsService
			 * @namespace tomtom.services
			 * @param {String} apiKey The API key used to access the Map Toolkit API.
			 */
			initialize: function(apiKey) {
				base.prototype.initialize.apply(this, arguments);
			},
			
			/**
			 * Gets the HD Traffic region information, which contains incident information at a country level.
			 * @method getRegions
			 * @param {Object} [options] Options to supply to the service.
			 * @param {String} [options.language=en] The two character ISO code of the output language to use for the regions and country names. Supported languages are af, cs, da, de, en, es, fi, fr, hu, it, nl, no, pl, pt, sv, tr.
			 * @param {String} [options.projection=EPSG4326] The map projection for the coordinates returned by the API. Supported projections are EPSG4326 and EPSG900913.
			 * @param {Function} callback The function to call when the service returns a response.
			 */
			getRegions: function(options, callback) {
				
				// make the options parameter optional
				if (typeof(options) == "function") {
					callback = options;
					options = {};
				}
				
				// if an API key is specified, include it
				// an API key can be ommitted if an API proxy is being used
				if (this.apiKey)
					options.key = this.apiKey;
				
				// check to see if an API proxy is being used
				// if so, use a regular XHR, JSONP will not work with a proxy
				if (tomtom.proxyUrl) {
					ajax.getJSON({
						url: utils.getServiceUrl(HDT_URL, options),
						callback: callback
					});
				} else {
					ajax.getJSONP({
						url: utils.getServiceUrl(HDT_URL + "p", options),
						callback: callback,
						callbackParam: "jsonp"
					});
				}
			}
			
		});
		
		var HDT_URL = "lbs/services/hdtRegions/3/json";
		
		return tomtom.services.HDTRegionsService;
	}
);
