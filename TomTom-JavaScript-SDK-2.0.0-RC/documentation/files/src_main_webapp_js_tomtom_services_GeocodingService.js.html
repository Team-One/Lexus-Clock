File: src/main/webapp/js/tomtom/services/GeocodingService.js

define([
        "./BaseService",
        "../Utils",
        "../AjaxUtil"
    ], 
    
    function(base, utils, ajax) {
	
		tomtom.services.GeocodingService = base.extend({
			
			/**
			 * A service which provides geocoding functionality.
			 * @constructor
			 * @class GeocodingService
			 * @namespace tomtom.services
			 * @param {String} apiKey The API key used to access the Map Toolkit API.
			 */
			initialize: function(apiKey) {
				base.prototype.initialize.apply(this, arguments);
			},
			
			/**
			 * Executes a geocoding request for the specified address.
			 * This will take an address and return the latitude and longitude of the address.
			 * @example
			 *	var service = new tomtom.services.GeocodingService("API Key");
			 *	service.geocode("11 Lafayette St, Lebanon, NH", function(response) { ... });
			 *
			 * @method geocode
			 * @param {String} address The address to geocode.
			 * @param {Object} [options] The options passed to the geocoding service.
			 * @param {String} [options.projection] The projection used to specify the coordinates in the request and response.  Supported values are
			 * 	EPSG4326 or EPSG900913.
			 * @param {String} [options.language] The two-character ISO code for the output language. Supported languages are de, en, es, fr, it, nl.
			 * @param {Number} [options.maxResults] The maximum number of hits for the geocoder to return.
			 * @param {Function} callback The callback to execute when a response is received from the geocoding service.
			 */
			geocode: function(address, options, callback) {
				
				// make the options parameter optional
				if (typeof(options) == "function") {
					callback = options;
					options = {};
				}
				
				var parameters = {
					q: address
				};
				
				// if an API key is specified, include it
				// an API key can be ommitted if an API proxy is being used
				if (this.apiKey)
					parameters.key = this.apiKey;
				
				// copy the options into the parameters
				parameters = L.Util.extend(parameters, options);

				// check to see if an API proxy is being used
				// if so, use a regular XHR, JSONP will not work with a proxy
				if (tomtom.proxyUrl) {
					ajax.getJSON({
						url: utils.getServiceUrl(GEOCODE_URL, parameters),
						callback: callback
					});
				} else {
					ajax.getJSONP({
						url: utils.getServiceUrl(GEOCODE_URL + "p", parameters),
						callback: callback,
						callbackParam: "jsonp"
					});
				}
			},
			
			/**
			 * Executes a reverse geocoding request, which allows a latitude and longitude to be turned into an address.
			 * @example
			 *	var service = new tomtom.services.GeocodingService("API Key");
			 *	service.reverseGeocode(43.67863, -72.26259, function(response) { ... });
			 *
			 * @method reverseGeocode
			 * @param {Number} latitude The latitude to look up the address for.
			 * @param {Number} longitude The longitude to look up the address for.
			 * @param {Object} [options] Additional options to pass to the geocoding service.
			 * @param {String} [options.type] Defines the importance of the returned value: are we looking for the name of a village (Regional), or the large city it's close to (International)? 
			 * Different values can result in different fields in the response, or different values for the same field.
			 * Basic (default) - Small number of fields. Contains formated address (concatenated data).
			 * All - List of results for all types together.
			 * International - Values of nearest big locations.
			 * National - Values of nearest medium locations.
			 * Regional - Values of nearest locations.
			 * @param {String} [options.projection] The projection used to specify the coordinates in the request and response.  Supported values are
			 * 	EPSG4326 or EPSG900913.
			 * @param {String} [options.language] The two-character ISO code for the output language. Supported languages are de, en, es, fr, it, nl.
			 * @param {Function} callback The callback to execute when a response is received from geocoding service.
			 */
			reverseGeocode: function(latitude, longitude, options, callback) {
				
				// check to see if the longitude was passed, if not assume the first param is a latlng object
				if (isNaN(longitude)) {
					callback = options;
					options = longitude;
					
					var point = L.latLng(latitude);
					latitude = point.lat;
					longitude = point.lng;
				}
				
				// make the options parameter optional
				if (typeof(options) == "function") {
					callback = options;
					options = {};
				}
				
				var parameters = {
					point: latitude + "," + longitude,
					projection: "EPSG4326"
				};
				
				// copy the options into parameters
				parameters = L.Util.extend(parameters, options);
				
				// if an API key is specified, include it
				// an API key can be ommitted if an API proxy is being used
				if (this.apiKey)
					parameters.key = this.apiKey;
				
				// check to see if an API proxy is being used
				if (tomtom.proxyUrl) {
					ajax.getJSON({
						url: utils.getServiceUrl(REVERSE_GEOCODE_URL, parameters),
						callback: callback
					});
				} else {
					ajax.getJSONP({
						url: utils.getServiceUrl(REVERSE_GEOCODE_URL + "p", parameters),
						callback: callback,
						callbackParam: "jsonp"
					});
				}
				
			}
			
		});
		
		var GEOCODE_URL = "lbs/services/geocode/3/json";
		var REVERSE_GEOCODE_URL = "lbs/services/reverseGeocode/3/json";
		
		return tomtom.services.GeocodingService;
	}
);
