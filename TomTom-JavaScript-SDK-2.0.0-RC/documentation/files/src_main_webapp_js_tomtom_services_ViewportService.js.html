File: src/main/webapp/js/tomtom/services/ViewportService.js

define([
        "./BaseService",
        "../AjaxUtil",
        "../Utils"
    ], 
    
    function(base, ajax, utils) {
	
		tomtom.services.ViewportService = base.extend({
			
			/**
			 * A service which gets information about the current viewport.
			 * @constructor
			 * @class ViewportService
			 * @namespace tomtom.services
			 * @param {String} apiKey The API key used to access the Map Toolkit API.
			 */
			initialize: function(apiKey) {
				base.prototype.initialize.apply(this, arguments);
			},
			
			/**
			 * Gets information about the current viewport including the current traffic model.
			 * This service call is used to synchronize different traffic components such as routing, traffic incidents and traffic flow using the
			 * traffic model id.
			 * @example
			 * 	var window = {
			 *		left: -78.40152658903608,
			 *		bottom: 39.19651962955681,
			 *		right: -59.054602760910434,
			 *		top: 44.89325482670453
			 *	}
			 *	
			 *	var service = new tomtom.services.ViewportService("API Key");
			 *	service.getViewportModel(window, 7, window, 7, function(response) { ... });
			 *
			 * @method getViewportModel
			 * @param {Object} window An object which describes the current map bounds.  This object should have left, bottom, right and top properties. 
			 * @param {Number} zoom The zoom level that the map is currently at.
			 * @param {Object} overviewWindow If an overview map is being used, this can be supplied to get viewport information for the
			 * 	overview map's bounds.  If an overview window is not used or copyright information isn't needed, the same bounds can be supplied as the main window.
			 * @param {Number} overviewZoom The zoom level of the overview map.  If there is no overview map, or copyright info isn't needed, the same zoom level as the main window
			 *  can be used.
			 * @param {Object} [options] Additional options to pass to the Viewport call.
			 * @param {String} [options.projection] The projection to be used for the bounding box.  Must be either EPSG4326 or EPSG900913.
			 * @param {Function} callback The callback to execute when the call to the Viewport service completes.
			 */
			getViewportModel: function(window, zoom, overviewWindow, overviewZoom, options, callback) {
				
				// make the options parameter optional
				if (typeof(options) == "function") {
					callback = options;
					options = {};
				}
				
				if (!options.projection)
					options.projection = "EPSG4326";
				
				var northEast = new L.LatLng(window.top, window.right, true);
				var southWest = new L.LatLng(window.bottom, window.left, true);
				var overviewNorthEast = new L.LatLng(overviewWindow.top, overviewWindow.right, true);
				var overviewSouthWest = new L.LatLng(overviewWindow.bottom, overviewWindow.left, true);
				
				// transform to EPSG900913
				if (options.projection == "EPSG4326") {
					
					var crs = L.CRS.EPSG3857;
					
					northEast = crs.project(northEast);
					southWest = crs.project(southWest);
					overviewNorthEast = crs.project(overviewNorthEast);
					overviewSouthWest = crs.project(overviewSouthWest);
					
				} else {
					
					northEast = new L.Point(northEast.lng, northEast.lat);
					southWest = new L.Point(southWest.lng, southWest.lat);
					overviewNorthEast = new L.Point(overviewNorthEast.lng, overviewNorthEast.lat);
					overviewSouthWest = new L.Point(overviewSouthWest.lng, overviewSouthWest.lat);
					
				}
				
				var mapMinX = southWest.x, mapMinY = southWest.y, mapMaxX = northEast.x, mapMaxY = northEast.y;
				
				mapMinX = (mapMinX < maximumBounds.minX) ? maximumBounds.minX : mapMinX;
				mapMinY = (mapMinY < maximumBounds.minY) ? maximumBounds.minY : mapMinY;
				mapMaxX = (mapMaxX > maximumBounds.maxX) ? maximumBounds.maxX : mapMaxX;
				mapMaxY = (mapMaxY > maximumBounds.maxY) ? maximumBounds.maxY : mapMaxY;

		        var ovMapMinX = (overviewSouthWest.x < maximumBounds.minX) ? maximumBounds.minX : overviewSouthWest.x;
		        var ovMapMinY = (overviewSouthWest.y < maximumBounds.minY) ? maximumBounds.minY : overviewSouthWest.y;
		        var ovMapMaxX = (overviewNorthEast.x > maximumBounds.maxX) ? maximumBounds.maxX : overviewNorthEast.x;
		        var ovMapMaxY = (overviewNorthEast.y > maximumBounds.maxY) ? maximumBounds.maxY : overviewNorthEast.y;
				
				var urlParameters = {
					box: mapMinY + "," + mapMinX + "," + mapMaxY + "," + mapMaxX,
					zoom: zoom,
					overviewBox: ovMapMinY + "," + ovMapMinX + "," + ovMapMaxY + "," + ovMapMaxX,
					overviewZoom: overviewZoom,
					copyright: typeof(options.copyright) == "undefined" ? true : options.copyright
				};
				
				// if an API key is specified, include it
				// an API key can be ommitted if an API proxy is being used
				if (this.apiKey)
					options.key = this.apiKey;
				
				// check to see if an API proxy is being used
				// if so, use a regular XHR, JSONP will not work with a proxy
				if (tomtom.proxyUrl) {
					ajax.getJSON({
						url: utils.getServiceUrl(L.Util.template(VIEWPORT_URL, urlParameters), options),
						callback: callback
					});
				} else {
					ajax.getJSONP({
						url: utils.getServiceUrl(L.Util.template(VIEWPORT_URL + "p", urlParameters), options),
						callback: callback,
						callbackParam: "jsonp"
					});
				}
			}
			
		});
		
		var maximumBounds = {
            "minX": -40075016.6855784,
            "minY": -40075016.6855784,
            "maxX": 40075016.6855784,
            "maxY": 40075016.6855784
        };
		
		var VIEWPORT_URL = "lbs/services/viewportDesc/3/{box}/{zoom}/{overviewBox}/{overviewZoom}/{copyright}/json";
		
		return tomtom.services.ViewportService;
	}
);
