define([
		"../Utils",
		"../dom/DomUtil"
	],
	
	function(utils, du) {
	
		tomtom.controls.ContextMenu = L.Control.extend({
			
			includes: L.Mixin.Events,
			
			/**
			 * Creates a new ContextMenu which can be used to display a menu to the user when a marker or the map is right clicked.
			 * @example
			 * 	map.setContextMenu(new tomtom.controls.ContextMenu({
			 * 		menuItems: [
			 * 			{ label: "Test", onClick: function() { alert("TEST!"); } }
			 * 		]
			 * 	}));
			 * @constructor
			 * @class ContextMenu
			 * @namespace tomtom.controls
			 * @extends L.Control
			 * @param {Object} options Options used to build the context menu.
			 * @param {Array} options.menuItems An array of the menu items to add.  
			 * 	Each object should have a label and optionally an onClick callback.
			 * 	See the example below. 
			 */			
			initialize: function(options) {
				
				// create the container
				this.container = document.createElement("ul");
				this._eventHandles = [];
				this._targetElements = [];
				this._menuItemNodes = [];
				
				if (options.menuItems) {
					
					this.menuItems = options.menuItems;
					
					for (var menuIndex = 0; menuIndex < this.menuItems.length; menuIndex++) {
						
						this.addMenuItem(this.menuItems[menuIndex]);
						
					}
					
				}
				
				/**
				 * Create a separate instance of the document click function.
				 * Leaflet's L.Util.stamp isn't very bright when creating unique identifiers for event handlers.
				 */
				this._documentClick = function(e) {
					
					// firefox hack, for some reason this event is fired when it shouldn't be
					if (e.button != 2)					
						this.hide();
					
				};
				
				// wire up the document click in order to hide the context menu when something else is clicked
				L.DomEvent.on(document, "click", this._documentClick, this);
				
			},
			
			/**
			 * Adds a new menu item to the context menu.
			 * @method addMenuItem
			 * @param {Object} menuItem The menu item to add to the context menu.
			 * @param {String} menuItem.label The label to be displayed to the user in the context menu.
			 * @param {Function} [menuItem.onClick] A callback to execute when the menu item is clicked.
			 */
			addMenuItem: function(menuItem) {
				
				var menuItemNode = document.createElement("li");
				
				menuItemNode.innerHTML = menuItem.label;
				
				// bind to the click method supplied
				L.DomEvent.on(menuItemNode, "click", function() { this._menuItemClick(menuItem); }, this);
				
				// handle the hover of the menu item to add a hover effect
				L.DomEvent.on(menuItemNode, "mouseover", this._menuItemHover);
				
				L.DomEvent.on(menuItemNode, "mouseout", this._menuItemBlur);
				
				this.container.appendChild(menuItemNode);
				
				// update the first/last elements
				du.query("li", this.container).removeClass("last");
				du.query("li", this.container).removeClass("first");
				du.query("li:last-child", this.container).addClass("last");
				du.query("li:first-child", this.container).addClass("first");
				
				this._menuItemNodes.push(menuItemNode);
				
			},
			
			_menuItemHover: function() {
				du.addClass(this, "menuItemHover");
			},

			_menuItemBlur: function() {
				du.removeClass(this, "menuItemHover");
			},
			
			_menuItemClick: function(menuItem) {
				
				var options = {
					menuItem: menuItem,
					contextMenu: this,
					x: this.currentX,
					y: this.currentY,
					pageX: this.currentX,
					pageY: this.currentY,
					targetElement: this.currentTargetElement,
					targetObject: this.currentTargetObject
				};
				
				// if the menu item has its own handler, fire it
				if (typeof(menuItem.onClick) == "function")
					menuItem.onClick(options);
				
				/**
				 * An event which is triggered when a menu item is clicked.
				 * @event menuitemclick
				 * @param {Object} eventArgs Arguments passed to the event that carry information about the event.
				 * @param {Object} eventArgs.menuItem The menu item which was clicked.
				 * @param {tomtom.ContextMenu} eventArgs.contextMenu The context menu that triggered the event.
				 * @param {Number} eventArgs.x The x coordinate of the current map relative position that the menu is being displayed at.
				 * @param {Number} eventArgs.y The y coordinate of the current map relative position that the menu is being displayed at.
				 * @param {DOM Node} eventArgs.targetElement The element for which the menu is being displayed.
				 * @param {Object} eventArgs.targetObject An object associated with the DOM Node for which the menu is being displayed.
				 */
				this.fire("menuitemclick", options);
				
			},
			
			/**
			 * Adds the context menu for the specified target element.
			 * A context menu can handle multiple targets.
			 * @method addTarget
			 * @param {DOM Node} targetElement The element to attach the context menu to.
			 * @param {Object} targetObject A data object to associate with the target element.
			 * 	This object will be passed in the menu item click events.
			 */
			addTarget: function(targetElement, targetObject) {
				
				if (this._targetElements.indexOf(targetElement) == -1) {
					
					targetElement._ttHandleContextMenu = L.Util.bind(function(e) {
						
						return this._onContextMenu(e, targetElement, targetObject);
						
					}, this);
				
					// attach to the context menu event
					L.DomEvent.on(targetElement, "contextmenu", targetElement._ttHandleContextMenu, this);
				
				}
				
			},
			
			/**
			 * Stops the context menu from displaying for the specified target.
			 * @method removeTarget
			 * @param {DOM Node} targetElement
			 */
			removeTarget: function(targetElement) {
				
				var index = -1;
				
				this._targetElements.forEach(function(element, currentIndex) {
					
					if (element == targetElement)
						index = currentIndex;
					
				});
				
				if (index != -1)
					this._targetElements.splice(index, 1);
				
				L.DomEvent.off(targetElement, "contextmenu", targetElement._ttHandleContextMenu, this);
				
			},
			
			/**
			 * Attaches custom data to the context menu.
			 * @method setData
			 * @param {Object} data The data to attach to the context menu.
			 */
			setData: function(data) {
				this.data = data;
			},
			
			/**
			 * Gets the data attached to this context menu.
			 * @method getData
			 * @returns Object
			 */
			getData: function() {
				return this.data;
			},
			
			onAdd: function(map) {
				
				this.map = map;
				
				du.addClass(this.container, "contextMenu");
				
				map._container.appendChild(this.container);
				
				if (!map._contextMenus)
					map._contextMenus = [];
				
				map._contextMenus.push(this);
				
			},
			
			/**
			 * Destroys the context menu and all associated resources.
			 * @method onRemove
			 */
			onRemove: function(map) {
				
				if (!map)
					map = this.map;
				
				L.DomEvent.off(document, "click", this._documentClick);
				this._documentClick = null;
				map._container.removeChild(this.container);
				
				this._targetElements.forEach(function(element) {
					this.removeTarget(element);
				}, this);
				
				// unbind each menu item
				this._menuItemNodes.forEach(function(menuItem) {
					du.off(menuItem, "mouseover", this._menuItemHover);
					du.off(menuItem, "mouseout", this._menuItemBlur);
					du.off(menuItem, "click", this._menuItemClick);
				}, this);
				
				this._menuItemNodes = [];
				this.menuItems = [];
				
			},
			
			/**
			 * Hides the context menu.
			 * @method hide 
			 */
			hide: function() {
				
				du.removeClass(this.container, "contextMenuActive");
				
			},
			
			_onContextMenu: function(e, targetElement, targetObject) {
				
				// get the offset of the map div
				var offset = du.offset(this.map._container);
				var containerSize = du.dimensions(this.container);
				var mapSize = du.dimensions(this.map._container);
				
				// calculate the x and y based on the offset
				var x = e.clientX - offset.left;
				var y = e.clientY - offset.top;
				
				// figure out if the menu fits in its current location
				if (x + containerSize.outerWidth > mapSize.width) {
					x -= containerSize.outerWidth;
				}
				
				if (y + containerSize.outerHeight > mapSize.height) {
					y -= containerSize.outerHeight;
				}
				
				this.currentX = x;
				this.currentY = y;
				
				// position the context menu
				du.css(this.container, "left", x + "px");
				du.css(this.container, "top", y + "px");
				
				du.addClass(this.container, "contextMenuActive");
				
				// make sure the event doesn't bubble up
				L.DomEvent.stop(e);
				
				this.currentTargetElement = targetElement;
				this.currentTargetObject = targetObject;
				
				return false;
				
			}
			
		});
		
		return tomtom.controls.ContextMenu;
	
	}
);
