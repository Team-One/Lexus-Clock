define([
		"../Utils",
		"../Main"
   ], 
    
    function(utils) {
	
		tomtom.layers.TrafficLayer = L.TileLayer.extend({
		    
			/**
			 * Displays traffic flow information on top of the TomTom base map.
			 * It is recommended that this class is not used directly but rather the {{#crossLink "tomtom.Map"}}Map{{/crossLink}}'s displayTraffic
			 * option is used like this:
			 * 
			 * 	var map = new tomtom.Map({
			 * 		domNode: "map",
			 * 		displayTraffic: true
			 * 	});
			 * 
			 * @constructor
			 * @class TrafficLayer
			 * @namespace tomtom.layers
			 * @extends L.TileLayer
			 * @param {String} apiKey The API key used to access the Map Toolkit API.
			 * @param {String} trafficModel The starting traffic model used to access the traffic service.
			 */
			initialize: function(apiKey, trafficModel) {
				
				if (typeof(options) == "undefined")
					options = {};
				
				var key = apiKey;
				
				L.TileLayer.prototype.initialize.apply(this, [
					TRAFFIC_TILE_URL, 
					{
						apiKey: key,
						trafficModel: trafficModel
					}
				]);
				
			},
		    
		    /**
		     * Updates the traffic layer using the specified traffic model.
		     * This will force all of the tiles to redraw.
		     * @method update
		     * @param {String} trafficModel The traffic model used to access the traffic service.
		     */
		    update: function(trafficModel) {
		    	
		    	// only redraw the layer if the traffic model changes
		    	if (this.options.trafficModel != trafficModel) {
		    		
			    	this.options.trafficModel = trafficModel;
			    	this.redraw();
			    	
		    	}
		    	
		    },
		    
		    /**
		     * Sets the traffic model that the layer will use when passed to the tile service.
		     * @method setTrafficModel
		     * @param {String} trafficModel
		     */
		    setTrafficModel: function(trafficModel) {
		    	this.options.trafficModel = trafficModel;
		    }
			
		});
		
		var TRAFFIC_TILE_URL = "http://{s}.api.tomtom.com/lbs/map/3/traffic/s2/{z}/{x}/{y}.png?t={trafficModel}&key={apiKey}";
		
		return tomtom.layers.TrafficLayer;
	
	}
);
