define([
        "./BaseService",
        "../AjaxUtil",
        "../Utils",
        "../i18n"
    ], 
	
	function(base, ajax, utils) {
		
		tomtom.services.TrafficService = base.extend({
			
			/**
			 * A service which provides traffic information via the Map Toolkit API.
			 * @constructor
			 * @class TrafficService
			 * @namespace tomtom.services
			 * @param {String} apiKey The API key used to access the Map Toolkit API.
			 */
			initialize: function(apiKey) {
				base.prototype.initialize.apply(this, arguments);
			},
			
			/**
			 * Gets traffic incident information for the specified window and zoom.
			 * The window should be specified as an object with left, bottom, right, and top properties.
			 * @example
			 *	var window = {
			 *		left: -78.40152658903608,
			 *		bottom: 39.19651962955681,
			 *		right: -59.054602760910434,
			 *		top: 44.89325482670453
			 *	}
			 *	
			 *	var trafficService = new tomtom.services.TrafficService("API Key");
			 *	trafficService.getTrafficModel(window, 7, function(response) { ... });
			 * 
			 * @method getTrafficModel
			 * @param {Object} window The bounds in which to retrieve the traffic information.
			 * @param {Number} zoom The zoom level at which to retrieve the traffic information.
			 * @param {Object} [options] Additional options to be passed to the traffic call.
			 * @param {String} [options.language=en] The two-character ISO code for the output language. Affects the &&lt;c&&gt; (cause) and &&lt;d&&gt; (description) fields. If this is not specified, those fields will not be included in the reply.
			 *	Supported languages are ca, cs, da, de, en, es, fi, fr, hu, it, nl, no, pl, pt, sk, sv, tr.
			 * @param {String} [options.projection=EPSG4326] The projection used to specify the coordinates in the request and response.
			 * @param callback
			 */
			getTrafficModel: function(window, zoom, options, callback) {
				
				// make the options parameter optional
				if (typeof(options) == "function") {
					callback = options;
					options = {};
				}
				
				// default the projection to EPSG 4326
				if (!options.projection)
					options.projection = "EPSG4326";
				
				// url parameters
				var parameters = L.Util.extend({}, options);
				
				// set the default traffic model ID
				if (!options.trafficModelID)
					options.trafficModelID = -1;
				
				// get rid of the traffic model from the parameters
				delete parameters.trafficModelID;
				
				// check to see if a language is specified
				if (!parameters.language)
					parameters.language = tomtom.LocaleManager.getPrimaryLanguage();
				
				options.zoom = zoom;
				options.minY = window.top;
				options.minX = window.left;
				options.maxY = window.bottom;
				options.maxX = window.right;
				
				// if an API key is specified, include it
				// an API key can be ommitted if an API proxy is being used
				if (this.apiKey)
					parameters.key = this.apiKey;
				
				parameters.expandCluster = true;
				
				// check to see if an API proxy is being used
				// if so, use a regular XHR, JSONP will not work with a proxy
				if (tomtom.proxyUrl) {
					ajax.getJSON({
						url: utils.getServiceUrl(L.Util.template(TRAFFIC_MODEL_URL, options), parameters),
						callback: callback
					});
				} else {
					ajax.getJSONP({
						url: utils.getServiceUrl(L.Util.template(TRAFFIC_MODEL_URL + "p", options), parameters),
						callback: callback,
						callbackParam: "jsonp"
					});
				}
				
			}
			
		});
		
		var TRAFFIC_MODEL_URL = "lbs/services/trafficIcons/3/s2/{minY},{minX},{maxY},{maxX}/{zoom}/{trafficModelID}/json";
		
		return tomtom.services.TrafficService;
	}
);
